# v3.51.0 MINOR — BLF (Vector Binary Logging Format) Parser

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add pure-.NET BLF (Vector Binary Logging Format) parsing to peakcan-host so users can load `.blf` traces captured with CANoe/CANalyzer without depending on the Vector XL Driver Library at runtime. Supports classic CAN 2.0, CAN FD, and CAN XL frame container types.

**Architecture:** Sister the v3.49.0 MINOR ASC parser pattern — a single static `BlfFormat` source-of-truth for constants and a `BlfParser/` directory of partial classes. `BlfParser.ParseAsync(Stream, ReplayOptions, ILogger?, CancellationToken)` returns `Task<IReadOnlyList<ReplayFrame>>` matching the existing `AscParser.ParseAsync` shape. `ReplayService.LoadAsync` dispatches by file extension (`.asc` → AscParser, `.blf` → BlfParser). Pure .NET — no Vector SDK, no Python/.NET interop — so peakcan-host parses BLF on any Windows machine.

**Tech Stack:** C# / .NET 10 / WPF (existing) / `BinaryReader` from `System.IO` for low-level byte access / xUnit (existing) for tests.

```csharp
namespace PeakCan.Host.Core.Replay;

/// <summary>
/// v3.51.0 MINOR: BLF (Vector Binary Logging Format) format single source.
/// Sister of v3.49.0 MINOR AscFormat. Pure static constants — no state,
/// no DI. All values verified against Vector BLF specification (constants
/// must be copied 1:1 from reverse-engineered reference per W22 + W23
/// LESSON; do not invent).
/// </summary>
public static class BlfFormat
{
    /// <summary>BLF file header signature: 4 ASCII bytes "LOGG".</summary>
    public const string FileSignature = "LOGG";

    /// <summary>BLF format signature (follows FileSignature): 4 ASCII bytes "LBLF".</summary>
    public const string FormatSignature = "LBLF";

    /// <summary>Supported version 1.0 (major 0x01, minor 0x00 packed in UINT32 LE).</summary>
    public const uint Version10 = 0x00010000;

    /// <summary>Supported version 2.0.</summary>
    public const uint Version20 = 0x00020000;

    /// <summary>Object header signature: 4 ASCII bytes "OBJH" (32-byte OBJH precedes each obj).</summary>
    public const string ObjHeader = "OBJH";

    /// <summary>Frame data blob signature: 4 ASCII bytes "BLOB".</summary>
    public const string Blob = "BLOB";

    /// <summary>Container object signature: 4 ASCII bytes "LOBJ" (wraps child objects).</summary>
    public const string Container = "LOBJ";

    /// <summary>Frame container type ID for classic CAN 2.0 (11/29-bit ID, 8-byte payload).</summary>
    public const uint ET_CAN_DATA = 5;

    /// <summary>Frame container type ID for CAN FD (flexible data-rate, up to 64-byte payload).</summary>
    public const uint ET_CAN_FD_DATA = 29;

    /// <summary>Frame container type ID for CAN XL (extended length, up to 2048-byte payload).</summary>
    public const uint ET_CAN_XL_DATA = 33;

    /// <summary>Classic CAN BLOB layout size in bytes: 2 (channel) + 2 (flags) + 1 (dlc) + 3 (reserved) + 4 (id) + 8 (data).</summary>
    public const int ClassicCanBlobSize = 20;

    /// <summary>CAN FD BLOB header size in bytes (data[64] follows): 20 (classic layout) + 1 (frameLength) + 3 (reserved) + 8 (padding before data[64]).</summary>
    public const int CanFdBlobSize = 32;

    /// <summary>CAN XL BLOB header size in bytes (data[2048] follows): 20 (classic layout) + 2 (frameLength BIG-ENDIAN) + 2 (reserved) + 8 (padding before data[2048]).</summary>
    public const int CanXlBlobMinSize = 32;

    /// <summary>OBJH timestamp is UINT64 in 100-nanosecond ticks; divide by this to get seconds.</summary>
    public const double TimestampScale = 10_000_000.0;

    /// <summary>Frame flag bit: CAN FD frame.</summary>
    public const ushort FlagFd = 0x0001;

    /// <summary>Frame flag bit: bit rate switch (CAN FD BRS).</summary>
    public const ushort FlagBrs = 0x0002;

    /// <summary>Frame flag bit: error state indicator (CAN FD ESI).</summary>
    public const ushort FlagEsi = 0x0004;

    /// <summary>Frame flag bit: CAN XL frame.</summary>
    public const ushort FlagXl = 0x0010;
}
```

```csharp
using FluentAssertions;
using Microsoft.Extensions.Logging.Abstractions;
using PeakCan.Host.Core.Replay;
using Xunit;

namespace PeakCan.Host.Core.Tests.Replay;

/// <summary>
/// v3.51.0 MINOR: verifies BlfParser.ParseAsync against synth BLF files
/// built with BinaryWriter. Manual test against a real CANoe-captured
/// BLF is in BlfParserManualTests (CI-skip).
/// </summary>
public class BlfParserTests

    private static ReplayOptions DefaultOptions() => new ReplayOptions();

    /// <summary>Build a synth BLF stream with the given OBJH/BLOB payloads.</summary>
    private static MemoryStream SynthBlf(params (string Signature, uint ObjType, byte[] Body)[] objects)
    {
        var ms = new MemoryStream();
        // File header: LOGG + LBLF + version UINT32 + app info
        ms.Write(Encoding.ASCII.GetBytes("LOGG"));
        ms.Write(Encoding.ASCII.GetBytes("LBLF"));
        ms.Write(BitConverter.GetBytes((uint)0x00010000)); // version 1.0 LE
        ms.Write(new byte[4]); // app info (ignored)
        // Objects
        foreach (var (sig, type, body) in objects)
        {
            ms.Write(Encoding.ASCII.GetBytes(sig));
            ms.Write(BitConverter.GetBytes((uint)body.Length));
            ms.Write(BitConverter.GetBytes(type));
            ms.Write(BitConverter.GetBytes((ulong)0)); // timestamp
            ms.Write(new byte[2]); // flags
            ms.Write(body);
        }
        return ms;
    }

    [Fact]
    public async Task BlfParser_ClassicCan_ValidBlob_Parsed()
    {
        // Classic CAN BLOB: 2 channel + 2 flags + 1 dlc + 3 reserved + 4 id + 8 data
        var body = new byte[20];
        BitConverter.GetBytes((ushort)1).CopyTo(body, 0);   // channel = 1
        BitConverter.GetBytes((ushort)0).CopyTo(body, 2);   // flags = 0
        body[4] = 8;                                          // dlc = 8
        BitConverter.GetBytes((uint)0x123).CopyTo(body, 8); // id = 0x123
        body[12] = 0xDE; body[13] = 0xAD; body[14] = 0xBE; body[15] = 0xEF;
        body[16] = 0x01; body[17] = 0x02; body[18] = 0x03; body[19] = 0x04;

        using var ms = SynthBlf(("BLOB", BlfFormat.ET_CAN_DATA, body));
        ms.Position = 0;
        var frames = await BlfParser.ParseAsync(ms, DefaultOptions());

        frames.Should().HaveCount(1);
        var f = frames[0];
        f.Id.Should().Be(0x123u);
        f.Dlc.Should().Be((byte)8);
        f.Flags.Should().Be(FrameFlags.None, "classic CAN has no FD/BRS/ESI/XL flags");
        f.Data.Should().Equal(0xDE, 0xAD, 0xBE, 0xEF, 0x01, 0x02, 0x03, 0x04);
    }

        BitConverter.GetBytes((ushort)BlfFormat.FlagFd).CopyTo(body, 2);
        body[4] = 16; // dlc = 16 (CAN FD scales to 16 even though up to 64 bytes)
        BitConverter.GetBytes((uint)0x456).CopyTo(body, 8);
        body[16] = 16; // frameLength = 16 bytes of actual data

        for (int i = 0; i < 16; i++) body[20 + i] = (byte)(0xA0 + i);

        // Override timestamp in OBJH header — find the offset
        ms.Position = 16; // skip file header (16 bytes)

        ms.Position = 0;

        ms.Position = 0;
        var frames = await BlfParser.ParseAsync(ms, DefaultOptions());

        frames[0].Data.Length.Should().Be(32,
            "CAN XL frame length is big-endian: bytes [0x00, 0x20] = 32, not 8192");
    }
```

```csharp
using Microsoft.Extensions.Logging;
using Microsoft.Extensions.Logging.Abstractions;

namespace PeakCan.Host.Core.Replay;

/// <summary>
/// v3.51.0 MINOR: parses Vector BLF trace files. Sister of v3.49.0
/// AscParser. Strict error handling (per user direction 2026-07-16):
/// bad magic → ReplayFormatException, >50% corrupted frames →
/// ReplayFormatException, truncated BLOB → ReplayFormatException.
/// Unknown object signatures are skipped with logger.Warning.
/// </summary>
public static partial class BlfParser
{
    private static ILogger _logger = NullLogger.Instance;

    [LoggerMessage(Level = LogLevel.Warning,
                   Message = "Skipped unknown BLF object signature '{Signature}' at offset {Offset}")]
    private static partial void LogUnknownObject(ILogger logger, string signature, long offset);

    [LoggerMessage(Level = LogLevel.Warning,
                   Message = "Skipped corrupted BLF frame at offset {Offset}: {Reason}")]
    private static partial void LogCorruptedFrame(ILogger logger, long offset, string reason);

    /// <summary>
    /// v3.51.0 MINOR: parse <paramref name="stream"/> as BLF. Sister of
    /// AscParser.ParseAsync. Throws ReplayFormatException on bad magic /
    /// version / &gt;50% corruption; throws ReplayLoadException on
    /// stream-size cap exceeded (via existing CountingStream path).
    /// </summary>
    public static async Task<IReadOnlyList<ReplayFrame>> ParseAsync(
        Stream stream,
        ReplayOptions options,
        ILogger? logger = null,
        CancellationToken ct = default)
    {
        ArgumentNullException.ThrowIfNull(options);
        _logger = logger ?? NullLogger.Instance;

        // File header precheck (sister of v3.10.0 T4 H5 AscParser precheck;
        // for simplicity we skip CountingStream wrap here and rely on
        // ReplayService.LoadAsync's precheck for seekable streams).
        if (stream.CanSeek && stream.Length < 16)
        {
            throw new ReplayFormatException($"BLF file too small: {stream.Length} bytes");
        }

        var result = new List<ReplayFrame>();
        int objectCount = 0;
        int errorCount = 0;

        using var reader = new BinaryReader(stream, System.Text.Encoding.ASCII, leaveOpen: true);

        // 1. File header (16 bytes)
        string fileSig = new string(reader.ReadChars(4));
        string formatSig = new string(reader.ReadChars(4));
        uint version = reader.ReadUInt32();
        reader.ReadUInt32(); // app info (ignored)

        if (fileSig != BlfFormat.FileSignature || formatSig != BlfFormat.FormatSignature)
        {
            throw new ReplayFormatException(
                $"Not a valid BLF file: bad magic '{fileSig}{formatSig}' (expected 'LOGGLBLF')");
        }
        if (version != BlfFormat.Version10 && version != BlfFormat.Version20)
        {
            throw new ReplayFormatException(
                $"Unsupported BLF version: 0x{version:X8} (expected 0x{BlfFormat.Version10:X8} or 0x{BlfFormat.Version20:X8})");
        }

        // 2. Object stream loop
        while (stream.Position < stream.Length)
        {
            ct.ThrowIfCancellationRequested();
            long objStart = stream.Position;
            string sig = new string(reader.ReadChars(4));
            uint objSize = reader.ReadUInt32();
            uint objType = reader.ReadUInt32();
            ulong timestamp = reader.ReadUInt64();
            ushort objFlags = reader.ReadUInt16();

            objectCount++;

            if (sig == BlfFormat.Blob)
            {
                long frameStart = stream.Position;
                try
                {
                    var frame = ParseBlobObject(objType, timestamp, objFlags, stream, objSize);
                    if (frame is not null) result.Add(frame);
                }
                catch (ReplayFormatException ex)
                {
                    errorCount++;
                    LogCorruptedFrame(_logger, frameStart, ex.Message);
                    // Skip remaining bytes of this BLOB and continue
                    long skipTo = objStart + 16 + objSize;
                    if (skipTo > stream.Length) break;
                    stream.Position = skipTo;
                }
            }
            else if (sig == BlfFormat.Container)
            {
                // LOBJ: container object. Frames are children; we read
                // them at child level. Skip container body.
                long skipTo = objStart + 16 + objSize;
                if (skipTo > stream.Length) break;
                stream.Position = skipTo;
            }
            else
            {
                LogUnknownObject(_logger, sig, objStart);
                long skipTo = objStart + 16 + objSize;
                if (skipTo > stream.Length) break;
                stream.Position = skipTo;
            }

            // 50% corruption threshold (sister of AscParser)
            if (objectCount > 1000 && errorCount * 2 > objectCount)
            {
                throw new ReplayFormatException(
                    $"BLF corruption: {errorCount}/{objectCount} objects failed (>{50}%)");
            }
        }

        if (result.Count == 0)
        {
            throw new ReplayFormatException("BLF file contains no parseable frames");
        }

        return result;
    }

    private static ReplayFrame? ParseBlobObject(
        uint objType, ulong timestamp, ushort objFlags, Stream stream, uint objSize)
    {
        return objType switch
        {
            BlfFormat.ET_CAN_DATA => ParseClassicCanBlob(timestamp, stream),
            BlfFormat.ET_CAN_FD_DATA => ParseCanFdBlob(timestamp, stream),
            BlfFormat.ET_CAN_XL_DATA => ParseCanXlBlob(timestamp, stream),
            _ => null, // unknown frame type — skip
        };
    }
}
```
