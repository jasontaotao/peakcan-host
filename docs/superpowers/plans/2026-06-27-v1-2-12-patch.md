private static partial void LogDeleteFromLibraryFailed(ILogger logger, Exception ex, string name);

{
    var saved = ToSavedFrame();
    try
    {
        _libraryService.Add(saved);
        RefreshLibrary();
        Status = $"Saved '{saved.Name}' to library ({_libraryService.Count} frames).";
    }
    catch (Exception ex)
    {
        Status = $"FAIL: Save '{saved.Name}' to library: {ex.Message}";
        if (_logger is not null) LogSaveToLibraryFailed(_logger, ex, saved.Name);
    }
}

public void DeleteFromLibrary(SavedFrame frame)
{
    if (frame is null) return;
    try
    {
        if (_libraryService.Remove(frame.Name))
        {
            RefreshLibrary();
            Status = $"Removed '{frame.Name}' from library.";
        }
        else
        {
            Status = $"'{frame.Name}' not found in library (already removed?).";
        }
    }
    catch (Exception ex)
    {
        Status = $"FAIL: Remove '{frame.Name}': {ex.Message}";
        if (_logger is not null) LogDeleteFromLibraryFailed(_logger, ex, frame.Name);
    }
}

using System.Threading.Channels;
// ... 其他 using

public sealed partial class RecordService : BackgroundService, ISink
{
    private readonly Channel<CanFrame> _frameChannel = Channel.CreateBounded<CanFrame>(
        new BoundedChannelOptions(8192)
        {
            FullMode = BoundedChannelFullMode.DropOldest,
            SingleReader = true,
            SingleWriter = false  // SDK 读线程 + UI 线程都可能写
        });

    private long _frameEnqueuedCount;
    private long _frameDroppedOnFullChannel;
    private long _frameCount;  // 已写入文件的帧数 (保留兼容)
    private volatile bool _isRecording;
    private volatile TextWriter? _writer;
    private readonly ILogger<RecordService> _logger;

    /// <summary>已 enqueue 到 channel 的帧数 (含被丢的)。</summary>
    public long FrameEnqueuedCount => Interlocked.Read(ref _frameEnqueuedCount);

    /// <summary>被 channel 满丢掉的帧数。</summary>
    public long FrameDroppedOnFullChannel => Interlocked.Read(ref _frameDroppedOnFullChannel);

    public void OnFrame(CanFrame frame)
    {
        if (!_isRecording || _writer is null) return;
        if (!_frameChannel.Writer.TryWrite(frame))
        {
            Interlocked.Increment(ref _frameDroppedOnFullChannel);
        }
        else
        {
            Interlocked.Increment(ref _frameEnqueuedCount);
        }
    }

    protected override async Task ExecuteAsync(CancellationToken stoppingToken)
    {
        var flushTimer = new PeriodicTimer(TimeSpan.FromSeconds(1));
        try
        {
            // Drain loop
            var drainTask = Task.Run(async () =>
            {
                try
                {
                    await foreach (var frame in _frameChannel.Reader.ReadAllAsync(stoppingToken))
                    {
                        try
                        {
                            WriteFrame(frame);
                            Interlocked.Increment(ref _frameCount);
                        }
                        catch (Exception ex)
                        {
                            LogFrameWriteFailed(_logger, ex);
                        }
                    }
                }
                catch (OperationCanceledException) { }
            }, stoppingToken);

            // Flush loop
            var flushTask = Task.Run(async () =>
            {
                try
                {
                    while (await flushTimer.WaitForNextTickAsync(stoppingToken))
                    {
                        try { _writer?.Flush(); } catch (Exception ex) { LogFrameWriteFailed(_logger, ex); }
                    }
                }
                catch (OperationCanceledException) { }
            }, stoppingToken);

            await Task.WhenAll(drainTask, flushTask);
        }
        finally
        {
            flushTimer.Dispose();
        }
    }

    // OnError — Item 11 fix
    public void OnError(Exception ex)
    {
        LogSinkError(_logger, ex, nameof(RecordService));
    }
}

[LoggerMessage(EventId = 6001, Level = LogLevel.Warning, Message = "{Service} OnError forwarded")]
private static partial void LogSinkError(ILogger logger, Exception ex, string service);
