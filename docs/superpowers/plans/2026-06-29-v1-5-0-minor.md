```csharp
using System;
using System.IO;

namespace PeakCan.Host.Core.Path;

/// <summary>
/// Defense-in-depth path validation. Rejects relative paths, traversal segments (`..`),
/// null bytes, and null/empty input. Returns canonical absolute path on success.
/// Does NOT restrict to specific root directories (deferred to v1.6.0 — see ADR-1).
/// </summary>
public static class PathNormalizer
{
    public static string Normalize(string? path)
    {
        if (path is null)
        {
            throw new PathNormalizationException(
                "Path is null.",
                attemptedPath: null,
                reason: PathNormalizationReason.NullPath);
        }

        if (path.Length == 0)
        {
            throw new PathNormalizationException(
                "Path is empty.",
                attemptedPath: path,
                reason: PathNormalizationReason.EmptyPath);
        }

        if (path.Contains('\0'))
        {
            throw new PathNormalizationException(
                $"Path contains null byte at index {path.IndexOf('\0')}.",
                attemptedPath: path,
                reason: PathNormalizationReason.NullByte);
        }

        // Relative path: doesn't start with drive letter (C:) or UNC (\\).
        if (!Path.IsPathRooted(path))
        {
            throw new PathNormalizationException(
                $"Path is not absolute (must start with drive letter or \\\\): {path}",
                attemptedPath: path,
                reason: PathNormalizationReason.RelativePath);
        }

        var canonical = Path.GetFullPath(path);

        // After canonicalization, check for traversal segments.
        // Split on both / and \ (Path.AltDirectorySeparatorChar + Path.DirectorySeparatorChar).
        var segments = canonical.Split(new[] { Path.DirectorySeparatorChar, Path.AltDirectorySeparatorChar }, StringSplitOptions.RemoveEmptyEntries);
        if (Array.IndexOf(segments, "..") >= 0)
        {
            throw new PathNormalizationException(
                $"Path contains traversal segment '..' after canonicalization: {canonical}",
                attemptedPath: path,
                reason: PathNormalizationReason.TraversalSegment);
        }

        return canonical;
    }
}
```

```csharp
            var handles = Array.Empty<string>();
            var result = new List<ChannelDescriptor>(handles.Length);
            foreach (var handle in handles)

                result.Add(new ChannelDescriptor(handle, $"PCAN-USB FD ({handle})"));
            }
            return result;
        }
        catch (Exception ex)

            return Array.Empty<ChannelDescriptor>();
        }

    {
        if (_disposed) return;
        _disposed = true;
```

```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using Microsoft.Extensions.Configuration;
using Microsoft.Extensions.Logging;
using PeakCan.Host.Core.Channel;
using ReactiveUI;

namespace PeakCan.Host.App.ViewModels;

/// <summary>
/// ViewModel for the channel picker ComboBox in the MainView header.
/// Loads available channels from <see cref="IChannelEnumerator"/> on ctor,
/// restores last-selected from <see cref="IConfiguration"/>, and raises
/// <see cref="ChannelChanged"/> when user picks a new channel.
/// </summary>
public sealed class ChannelPickerViewModel : ReactiveObject
{
    private readonly IConfiguration _configuration;
    private readonly ILogger<ChannelPickerViewModel> _logger;

    public IReadOnlyList<ChannelDescriptor> AvailableChannels { get; }

    [Reactive]
    public ChannelDescriptor? SelectedChannel { get; set; }

    [Reactive]
    public string? SelectedChannelError { get; set; }

    public event EventHandler<ChannelDescriptor?>? ChannelChanged;

    public ChannelPickerViewModel(
        IChannelEnumerator enumerator,
        IConfiguration configuration,
        ILogger<ChannelPickerViewModel> logger)
    {
        _configuration = configuration;
        _logger = logger;
        AvailableChannels = enumerator.Enumerate();

        // Restore last-selected from configuration if present.
        var lastHandle = _configuration["Channel:SelectedHandle"];
        if (!string.IsNullOrEmpty(lastHandle))
        {
            SelectedChannel = AvailableChannels.FirstOrDefault(c => c.Handle == lastHandle);
            if (SelectedChannel is null)
            {
                _logger.LogWarning(
                    "Last-selected channel '{LastHandle}' not found in available channels; defaulting to none.",
                    lastHandle);
            }
        }

        this.WhenAnyValue(x => x.SelectedChannel).Subscribe(channel =>
        {
            // Persist to configuration.
            _configuration["Channel:SelectedHandle"] = channel?.Handle;
            // Raise event.
            ChannelChanged?.Invoke(this, channel);
        });
    }
}
```

```csharp
public bool Loop
{
    get => _replayService.Loop;
    set

        if (_replayService.Loop != value)

            _replayService.Loop = value;
            this.RaisePropertyChanged();

[Reactive]
public string CanIdFilterText { get; set; } = string.Empty;

[Reactive]
public string? CanIdFilterError { get; set; }

partial void OnCanIdFilterTextChanged(string value)
{
    if (string.IsNullOrWhiteSpace(value))
    {
        _replayService.CanIdFilter = null;
        CanIdFilterError = null;
        return;
    }

    var tokens = value.Split(new[] { ',', ' ', '\t', '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
    var ids = new HashSet<uint>();
    var errors = new List<string>();
    foreach (var token in tokens)
    {
        var trimmed = token.Trim();
        if (trimmed.StartsWith("0x", StringComparison.OrdinalIgnoreCase))
        {
            if (uint.TryParse(trimmed.AsSpan(2), System.Globalization.NumberStyles.HexNumber, null, out var id))
            {
                ids.Add(id);
            }
            else
            {
                errors.Add(trimmed);
            }
        }
        else
        {
            if (uint.TryParse(trimmed, out var id))
            {
                ids.Add(id);
            }
            else
            {
                errors.Add(trimmed);
            }
        }
    }

    _replayService.CanIdFilter = ids.Count > 0 ? ids : new HashSet<uint>();
    CanIdFilterError = errors.Count > 0 ? $"Invalid token(s): {string.Join(", ", errors)}" : null;
}
```
