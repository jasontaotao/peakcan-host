```csharp
[Fact]
public async Task Load_With_CancelledToken_Returns_Cancelled_Code()
{
    // Arrange — real DbcService + pre-cancelled CancellationToken. This
    // exercises the production silent-cancel branch at
    // DbcService.cs:162-165 (catch (OperationCanceledException) { })
    // combined with DbcApi.Load's post-load check at DbcApi.cs:115-121
    // (returns errorCode="Cancelled" when no DbcLoaded / LoadFailed
    // fired). Mirrors the FakeDbcService silent-cancel test above but
    // uses a real cancellation token instead of a test double.
    var dbcService = new DbcService(NullLogger<DbcService>.Instance);
    var api = new DbcApi(NullLogger<DbcApi>.Instance, dbcService);
    using var cts = new CancellationTokenSource();
    cts.Cancel();

    // Act
    var r = Unload(await api.Load("/any/path", cts.Token));

    // Assert
    r.Success.Should().BeFalse();
    r.MessageCount.Should().Be(0);
    r.ErrorCode.Should().Be("Cancelled");
    r.Error.Should().Be("Load was cancelled");
}
```
